cd "./ark-audio-gen"

zip -r deploy.zip . \
  -x ".venv/*" \
  -x "generated/*" \
  -x ".git/*" \
  -x "__pycache__/*" \
  -x "*.pyc" \
  -x "*.pyo" \
  -x ".DS_Store" \
  -x "*.mp3" \
  -x "*.wav" \
  -x "*.flac" \
  -x "deploy.zip"

# Confirm size (should be < 200 KB — no model weights, no venv)
ls -lh deploy.zip

deploy the zip
az webapp deploy \
  --resource-group rg-cad-planner-i1-d \
  --name ntt-sap-tkt \
  --slot audio-gen \
  --src-path deploy.zip \
  --type zip \
  --async false
